    const contentTitle = document.getElementById(contentTitle);

    const contentTagline = document.getElementById(contentTagline);

    const contentBody = document.getElementById(contentBody);

 

    function loadSection(key) {

      const section = sections[key];

      if (!section) return;

 

      contentTitle.textContent = section.title;

      contentTagline.textContent = section.tagline;

      contentBody.innerHTML = section.body;

 

      document.querySelectorAll(.guide-card).forEach(card = {

        card.classList.remove(ring-2, ring-emerald-400, ring-offset-2, ring-offset-slate-950);

      });

      const activeCard = document.querySelector(`.guide-card[data-section=${key}]`);

      if (activeCard) {

        activeCard.classList.add(ring-2, ring-emerald-400, ring-offset-2, ring-offset-slate-950);

      }

    }

 

    document.querySelectorAll(.guide-card).forEach(card = {

      card.addEventListener(click, () = {

        const key = card.getAttribute(data-section);

        loadSection(key);

      });

    });

 

    document.querySelectorAll(.nav-btn).forEach(btn = {

      btn.addEventListener(click, () = {

        const key = btn.getAttribute(data-section);

        if (key) {

          loadSection(key);

          window.scrollTo({ top 0, behavior smooth });

        }

      });

    });
